pwd							--显示当前目录
ls							--列出当前目录下文件/文件夹
git init					--初始化一个Git仓库（当前目录）
git add <file>				--添加文件Git仓库
git commit -m"xxx"			--提交文件Git仓库 -m"description"
git status					--查看工作区状态
git diff					--查看修改内容
git reset --hard commit_id	--在版本间穿梭
git reset --hard HEAD^		--回退到上一个版本
git log						--查看提交日志
git log --pretty=oneline	--查看提交日志（仅commit_id和说明）
git reflog					--查看命令记录
cat <file>					--查看文件内容
git checkout -- <file>		--撤销工作区的修改
git reset HEAD <file>		--撤销暂存区的修改（工作区中修改仍在）
rm <file>					--将文件删除
git rm <file>				--将文件从版本库删除
